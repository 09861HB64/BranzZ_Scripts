    CardCorner.Parent = Card

    local CardStroke = Instance.new("UIStroke")
    CardStroke.Color = Color3.fromRGB(40, 40, 55)
    CardStroke.Thickness = 1
    CardStroke.Parent = Card

    -- Ícone lateral
    local IconBg = Instance.new("Frame")
    IconBg.Size = UDim2.new(0, 38, 0, 38)
    IconBg.Position = UDim2.new(0, 12, 0.5, -19)
    IconBg.BackgroundColor3 = Color3.fromRGB(88, 101, 242)
    IconBg.BorderSizePixel = 0
    IconBg.Parent = Card
    local IconCorner = Instance.new("UICorner")
    IconCorner.CornerRadius = UDim.new(0, 10)
    IconCorner.Parent = IconBg

    local IconLabel = Instance.new("TextLabel")

    IconLabel.BackgroundTransparency = 1

    IconLabel.TextSize = 18

    FuncName.TextXAlignment = Enum.TextXAlignment.Left
    FuncName.Parent = Card

    -- Descrição
    local FuncDesc = Instance.new("TextLabel")
    FuncDesc.Size = UDim2.new(1, -130, 0, 18)
    FuncDesc.Position = UDim2.new(0, 60, 0, 35)
    FuncDesc.BackgroundTransparency = 1

    FuncDesc.TextSize = 11
    FuncDesc.Font = Enum.Font.Gotham
    FuncDesc.TextXAlignment = Enum.TextXAlignment.Left
    FuncDesc.Parent = Card

    local ToggleBg = Instance.new("Frame")
    ToggleBg.Size = UDim2.new(0, 52, 0, 30)
    ToggleBg.Position = UDim2.new(1, -64, 0.5, -15)
    ToggleBg.BackgroundColor3 = Color3.fromRGB(50, 50, 65)

    ToggleBgCorner.CornerRadius = UDim.new(1, 0)
    ToggleBgCorner.Parent = ToggleBg

    local ToggleCircle = Instance.new("Frame")
    ToggleCircle.Size = UDim2.new(0, 24, 0, 24)

    ToggleCircle.BackgroundColor3 = Color3.fromRGB(210, 210, 220)
    ToggleCircle.BorderSizePixel = 0
    ToggleCircle.Parent = ToggleBg
    local CircleCorner = Instance.new("UICorner")
    CircleCorner.CornerRadius = UDim.new(1, 0)
    CircleCorner.Parent = ToggleCircle

    -- Sombra no círculo
    local CircleShadow = Instance.new("UIStroke")
    CircleShadow.Color = Color3.fromRGB(0, 0, 0)
    CircleShadow.Thickness = 0.5
    CircleShadow.Transparency = 0.7
    CircleShadow.Parent = ToggleCircle

    -- Botão invisível sobre o toggle
    local ToggleBtn = Instance.new("TextButton")
    ToggleBtn.Size = UDim2.new(1, 0, 1, 0)
    ToggleBtn.BackgroundTransparency = 1
    ToggleBtn.Text = ""
    ToggleBtn.Parent = ToggleBg

    local isOn = false

    local function setToggle(state, animate)
        isOn = state
        local targetPos = state and UDim2.new(0, 25, 0.5, -12) or UDim2.new(0, 3, 0.5, -12)
        local targetColor = state and Color3.fromRGB(48, 209, 88) or Color3.fromRGB(50, 50, 65)
        local circleColor = state and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(210, 210, 220)

        if animate then
            TweenService:Create(ToggleCircle, TweenInfo.new(0.25, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                Position = targetPos,
                BackgroundColor3 = circleColor
            }):Play()
            TweenService:Create(ToggleBg, TweenInfo.new(0.25, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                BackgroundColor3 = targetColor
            }):Play()
        else
            ToggleCircle.Position = targetPos
            ToggleCircle.BackgroundColor3 = circleColor
            ToggleBg.BackgroundColor3 = targetColor
        end
    end

    return Card, ToggleBtn, setToggle, function() return isOn end
end

local function ativarTeleguiad()
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChild("Humanoid")
    if not hrp or not hum then return end

    hum.PlatformStand = true

    local bv = Instance.new("BodyVelocity")
    bv.Name = "TeleguiadBV"
    bv.MaxForce = Vector3.new(1e5, 1e5, 1e5)
    bv.Velocity = Vector3.new(0, 0, 0)
    bv.Parent = hrp

    teleguiadConnection = RunService.Heartbeat:Connect(function()
        if not teleguiadAtivo then return end
        local char2 = player.Character
        if not char2 then return end
        local hrp2 = char2:FindFirstChild("HumanoidRootPart")
        local bv2 = hrp2 and hrp2:FindFirstChild("TeleguiadBV")
        if not hrp2 or not bv2 then return end

        local cam = workspace.CurrentCamera
        local dir = cam.CFrame.LookVector
        bv2.Velocity = dir * 26.7
    end)
end

local function desativarTeleguiad()
    if teleguiadConnection then
        teleguiadConnection:Disconnect()
        teleguiadConnection = nil
    end
    local char = player.Character
    if char then
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            local bv = hrp:FindFirstChild("TeleguiadBV")
            if bv then bv:Destroy() end
        end
        local hum = char:FindFirstChild("Humanoid")
        if hum then hum.PlatformStand = false end
    end
end

TeleguiadBtn.MouseButton1Click:Connect(function()
    teleguiadAtivo = not teleguiadAtivo
    setTeleguiad(teleguiadAtivo, true)

    if teleguiadAtivo then
        ativarTeleguiad()
    else
        desativarTeleguiad()
    end
end)

-- Reativar ao respawnar
player.CharacterAdded:Connect(function()
    wait(0.5)
    if teleguiadAtivo then
        ativarTeleguiad()
    end
end)
