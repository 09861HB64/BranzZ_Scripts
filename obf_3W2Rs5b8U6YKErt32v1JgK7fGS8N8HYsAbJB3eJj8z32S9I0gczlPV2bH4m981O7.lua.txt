-- ============================================
--   Script de Notificação de Atualização
--   Discord: https://discord.gg/4RmZv2GyBx
-- ============================================

local DISCORD_LINK = "https://discord.gg/4RmZv2GyBx"

-- Verifica se está rodando no ambiente Roblox
if game and Players then
    -- ========================
    --  VERSÃO ROBLOX (LocalScript)
    -- ========================
    local Players = game:GetService("Players")
    local localPlayer = Players.LocalPlayer
    local playerGui = localPlayer:WaitForChild("PlayerGui")

    -- Cria o ScreenGui
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "UpdateNotification"
    screenGui.ResetOnSpawn = false
    screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    screenGui.Parent = playerGui

    -- Fundo escurecido
    local overlay = Instance.new("Frame")
    overlay.Size = UDim2.new(1, 0, 1, 0)
    overlay.Position = UDim2.new(0, 0, 0, 0)
    overlay.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    overlay.BackgroundTransparency = 0.4
    overlay.ZIndex = 10
    overlay.Parent = screenGui

    -- Janela principal
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 420, 0, 260)
    frame.Position = UDim2.new(0.5, -210, 0.5, -130)
    frame.BackgroundColor3 = Color3.fromRGB(30, 30, 40)
    frame.BorderSizePixel = 0
    frame.ZIndex = 11
    frame.Parent = screenGui

    -- Arredondamento da janela
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 12)
    corner.Parent = frame

    -- Barra do topo (título)
    local topBar = Instance.new("Frame")
    topBar.Size = UDim2.new(1, 0, 0, 45)
    topBar.BackgroundColor3 = Color3.fromRGB(88, 101, 242) -- Azul Discord
    topBar.BorderSizePixel = 0
    topBar.ZIndex = 12
    topBar.Parent = frame

    local topCorner = Instance.new("UICorner")
    topCorner.CornerRadius = UDim.new(0, 12)
    topCorner.Parent = topBar

    -- Fix canto inferior do topBar
    local topBarFix = Instance.new("Frame")
    topBarFix.Size = UDim2.new(1, 0, 0, 12)
    topBarFix.Position = UDim2.new(0, 0, 1, -12)
    topBarFix.BackgroundColor3 = Color3.fromRGB(88, 101, 242)
    topBarFix.BorderSizePixel = 0
    topBarFix.ZIndex = 12
    topBarFix.Parent = topBar

    -- Título
    local title = Instance.new("TextLabel")
    title.Size = UDim2.new(1, -50, 1, 0)
    title.Position = UDim2.new(0, 15, 0, 0)
    title.BackgroundTransparency = 1
    title.Text = "⚠️  Atualização Disponível"
    title.TextColor3 = Color3.fromRGB(255, 255, 255)
    title.TextSize = 17
    title.Font = Enum.Font.GothamBold
    title.TextXAlignment = Enum.TextXAlignment.Left
    title.ZIndex = 13
    title.Parent = topBar

    -- Botão fechar (X)
    local closeBtn = Instance.new("TextButton")
    closeBtn.Size = UDim2.new(0, 35, 0, 35)
    closeBtn.Position = UDim2.new(1, -42, 0, 5)
    closeBtn.BackgroundColor3 = Color3.fromRGB(220, 60, 60)
    closeBtn.Text = "✕"
    closeBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeBtn.TextSize = 16
    closeBtn.Font = Enum.Font.GothamBold
    closeBtn.BorderSizePixel = 0
    closeBtn.ZIndex = 14
    closeBtn.Parent = topBar

    local closeCorner = Instance.new("UICorner")
    closeCorner.CornerRadius = UDim.new(0, 8)
    closeCorner.Parent = closeBtn

    -- Mensagem principal
    local message = Instance.new("TextLabel")
    message.Size = UDim2.new(1, -30, 0, 80)
    message.Position = UDim2.new(0, 15, 0, 55)
    message.BackgroundTransparency = 1
    message.Text = "🔔 Esse script foi Atualizado!\n\nPor favor entre no servidor do Discord\npara pegar o novo!"
    message.TextColor3 = Color3.fromRGB(220, 220, 235)
    message.TextSize = 15
    message.Font = Enum.Font.Gotham
    message.TextXAlignment = Enum.TextXAlignment.Left
    message.TextYAlignment = Enum.TextYAlignment.Top
    message.TextWrapped = true
    message.ZIndex = 12
    message.Parent = frame

    -- Label do link
    local linkLabel = Instance.new("TextLabel")
    linkLabel.Size = UDim2.new(1, -30, 0, 20)
    linkLabel.Position = UDim2.new(0, 15, 0, 145)
    linkLabel.BackgroundTransparency = 1
    linkLabel.Text = "🔗 Link do Discord:"
    linkLabel.TextColor3 = Color3.fromRGB(160, 160, 180)
    linkLabel.TextSize = 13
    linkLabel.Font = Enum.Font.GothamBold
    linkLabel.TextXAlignment = Enum.TextXAlignment.Left
    linkLabel.ZIndex = 12
    linkLabel.Parent = frame

    -- Caixa do link
    local linkBox = Instance.new("TextBox")
    linkBox.Size = UDim2.new(1, -30, 0, 36)
    linkBox.Position = UDim2.new(0, 15, 0, 167)
    linkBox.BackgroundColor3 = Color3.fromRGB(20, 20, 30)
    linkBox.Text = DISCORD_LINK
    linkBox.TextColor3 = Color3.fromRGB(88, 180, 242)
    linkBox.TextSize = 13
    linkBox.Font = Enum.Font.Gotham
    linkBox.BorderSizePixel = 0
    linkBox.ClearTextOnFocus = false
    linkBox.ZIndex = 12
    linkBox.Parent = frame

    local linkBoxCorner = Instance.new("UICorner")
    linkBoxCorner.CornerRadius = UDim.new(0, 8)
    linkBoxCorner.Parent = linkBox

    -- Botão copiar link
    local copyBtn = Instance.new("TextButton")
    copyBtn.Size = UDim2.new(1, -30, 0, 38)
    copyBtn.Position = UDim2.new(0, 15, 0, 210)
    copyBtn.BackgroundColor3 = Color3.fromRGB(88, 101, 242)
    copyBtn.Text = "📋  Copiar Link"
    copyBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    copyBtn.TextSize = 15
    copyBtn.Font = Enum.Font.GothamBold
    copyBtn.BorderSizePixel = 0
    copyBtn.ZIndex = 12
    copyBtn.Parent = frame

    local copyCorner = Instance.new("UICorner")
    copyCorner.CornerRadius = UDim.new(0, 8)
    copyCorner.Parent = copyBtn

    -- Ação do botão fechar
    closeBtn.MouseButton1Click:Connect(function()
        screenGui:Destroy()
    end)

    -- Ação do botão copiar
    copyBtn.MouseButton1Click:Connect(function()
        -- Tenta copiar via setclipboard (funciona em alguns executores)
        if setclipboard then
            setclipboard(DISCORD_LINK)
            copyBtn.Text = "✅  Copiado!"
            copyBtn.BackgroundColor3 = Color3.fromRGB(50, 180, 100)
        else
            -- Seleciona o texto da caixa como fallback
            linkBox:CaptureFocus()
            linkBox.SelectionStart = 1
            linkBox.CursorPosition = #DISCORD_LINK + 1
            copyBtn.Text = "✅  Selecione e Ctrl+C"
            copyBtn.BackgroundColor3 = Color3.fromRGB(50, 180, 100)
        end

        -- Volta ao texto original após 2.5 segundos
        task.delay(2.5, function()
            if copyBtn and copyBtn.Parent then
                copyBtn.Text = "📋  Copiar Link"
                copyBtn.BackgroundColor3 = Color3.fromRGB(88, 101, 242)
            end
        end)
    end)

else
    -- ========================
    --  VERSÃO LOVE2D / LUA PURO
    -- ========================
    print("========================================")
    print("  ⚠️  SCRIPT ATUALIZADO!")
    print("========================================")
    print("")
    print("  Esse script foi Atualizado!")
    print("  Por favor entre no servidor do Discord")
    print("  para pegar o novo!")
    print("")
    print("  🔗 Link: " .. DISCORD_LINK)
    print("")
    print("========================================")

    -- Tenta abrir o clipboard se disponível
    if io then
        local ok, _ = pcall(function()
            local f = io.popen("echo " .. DISCORD_LINK .. " | xclip -selection clipboard 2>/dev/null || echo " .. DISCORD_LINK .. " | pbcopy 2>/dev/null", "w")
            if f then f:close() end
        end)
    end
end
