-- Cache da base — reavalia a cada 10s
local baseRef=nil;local baseStamp=0
local collectParts=nil

local function findBase()
    if baseRef and baseRef.Parent and (tick()-baseStamp)<10 then return baseRef end
    baseRef=nil;collectParts=nil
    local sv=workspace:FindFirstChild("Server") or workspace:FindFirstChild("Serve")
    if not sv then return nil end
    local bs=sv:FindFirstChild("Bases");if not bs then return nil end
    local myName=lp.Name
    for _,b in ipairs(bs:GetChildren()) do
        -- Tenta achar o nome diretamente no Sign sem GetDescendants pesado
        local found=false
        pcall(function()
            local sign=b:FindFirstChild("Sign",true)
            if not sign then return end
            -- Procura so nos filhos diretos do sign primeiro (mais rapido)
            for _,d in ipairs(sign:GetChildren()) do
                if (d:IsA("TextLabel") or d:IsA("TextBox")) and tostring(d.Text):find(myName,1,true) then
                    found=true;return
                end
            end
            -- Fallback: GetDescendants so se nao achou
            if not found then
                for _,d in ipairs(sign:GetDescendants()) do
                    if (d:IsA("TextLabel") or d:IsA("TextBox")) and tostring(d.Text):find(myName,1,true) then
                        found=true;return
                    end
                end
            end
        end)
        if found then baseRef=b;baseStamp=tick();return b end
    end
    return nil
end

local function getCollectParts()
    if collectParts then
        -- Verifica se as parts ainda existem
        if collectParts[1] and collectParts[1].Parent then return collectParts end
        collectParts=nil
    end
    local base=findBase();if not base then return nil end
    local slots=base:FindFirstChild("Slots",true);if not slots then return nil end
    local parts={}
    for _,s in ipairs(slots:GetDescendants()) do
        if s.Name=="Collect" and s:IsA("BasePart") then
            table.insert(parts,s)
        end
    end
    collectParts=#parts>0 and parts or nil
    return collectParts
end

-- LOOP PRINCIPAL — 0.5s entre ciclos completos — zero lag
-- firetouchinterest e instantaneo no servidor, nao precisa ser rapido
local function runLoop()
    while on do
        local parts=getCollectParts()
        if not parts then
            stLbl.Text="Base nao encontrada"
            stLbl.TextColor3=Color3.fromRGB(255,80,80)
            task.wait(3);continue
        end
        -- Dispara todos os toques de uma vez (sem yield entre eles)
        for _,s in ipairs(parts) do
            if not on then break end
            if not s.Parent then collectParts=nil;break end
            pcall(firetouchinterest,r,s,0)
            pcall(firetouchinterest,r,s,1)
        end
        collectCount=collectCount+1
        if collectCount%4==0 then
            stLbl.Text=("Coletando... #%d"):format(collectCount)
            stLbl.TextColor3=Color3.fromRGB(48,209,88)
        end
        task.wait(0.5)  -- 0.5s = sem lag, ainda coleta tudo
    end
end

local function setOn(v)
    on=v;collectCount=0
    if v then
        twnT(track,"BackgroundColor3",Color3.fromRGB(48,209,88))
        twnT(knob,"Position",UDim2.new(0,22,0,2))
        togLbl.Text="ON - coletando";togLbl.TextColor3=Color3.fromRGB(48,209,88)
        stLbl.Text="Ativo! Ande livremente"
        stLbl.TextColor3=Color3.fromRGB(48,209,88)
        if collectThread then pcall(task.cancel,collectThread) end
        collectThread=task.spawn(runLoop)
    else
        twnT(track,"BackgroundColor3",Color3.fromRGB(58,58,60))
        twnT(knob,"Position",UDim2.new(0,2,0,2))
        togLbl.Text="OFF - inativo";togLbl.TextColor3=Color3.fromRGB(140,140,155)
        stLbl.Text="Aguardando ativacao..."
        stLbl.TextColor3=Color3.fromRGB(100,100,120)
        if collectThread then pcall(task.cancel,collectThread);collectThread=nil end
    end
end
